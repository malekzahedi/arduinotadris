#include <LiquidCrystal.h>

// نعریف شیء از کتابخانه
LiquidCrystal myCLCD(12,11,5,4,3,2);
String reading;

void setup() {
  Serial.begin(9600);
  //شروع ارتباط با ماژول
  myCLCD.begin(16,2);
  //پاک کردن ال سی دی
  myCLCD.clear();
  //قرار دادن نشانگر در نقطه ی مورد نظر
  myCLCD.setCursor(0,0);
}

void loop() {
  while(Serial.available()){
    reading = Serial.read();
    Serial.flush();  
  }
  //پرینت
  myCLCD.print(reading);
}
