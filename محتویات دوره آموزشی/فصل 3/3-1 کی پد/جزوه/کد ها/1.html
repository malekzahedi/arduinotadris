// فراخواندن کتابخانه
#include <Keypad.h>

const byte ROWS = 4; // تعداد سطر
const byte COLS = 3; // تعداد ستون
// تعيين آرايش کيپد
char keys[ROWS][COLS] = {
  {'1','2','3'},
  {'4','5','6'},
  {'7','8','9'},
  {'#','0','*'}
};
// تعيين اتصال سطرهاي 0 تا 3 به پين هاي 2 تا 5 آردوينو
byte rowPins[ROWS] = { 2, 3, 4, 5};
// تعيين اتصال ستون هاي 0 تا 2 به پين هاي 6 تا 8 آردوينو
byte colPins[COLS] = { 6, 7, 8}; 

// درست کردن متغير کيپد
Keypad kpd = Keypad( makeKeymap(keys), rowPins, colPins, ROWS, COLS );


void setup()
{
  pinMode(ledpin,OUTPUT);
  Serial.begin(9600);
}

void loop()
{
  //گرفتن کليد
  char key = kpd.getKey();
 if (key)
  Serial.println(key);
}
