#define S0 4
#define S1 5
#define S2 6
#define S3 7
#define sensorOut 8

int period = 0;

void setup() {
  pinMode(S0, OUTPUT);
  pinMode(S1, OUTPUT);
  pinMode(S2, OUTPUT);
  pinMode(S3, OUTPUT);
  pinMode(sensorOut, INPUT);
  digitalWrite(S0,HIGH);
  digitalWrite(S1,HIGH);
  Serial.begin(9600);
}

void loop() {
  // تنظیمات برای خواندن رنگ قرمز
  digitalWrite(S2,LOW);
  digitalWrite(S3,LOW);
  // خواندن زمان تناوب
  period = pulseIn(sensorOut, LOW);
  Serial.print("R= ");
  Serial.print(period);
  Serial.print("  ");
  delay(100);

  // تنظیمات برای خواندن رنگ سبز
  digitalWrite(S2,HIGH);
  digitalWrite(S3,HIGH);
  period = pulseIn(sensorOut, LOW);
  Serial.print("G= ");
  Serial.print(period);
  Serial.print("  ");
  delay(100);

// تنظیمات برای خواندن رنگ آبی  
  digitalWrite(S2,LOW);
  digitalWrite(S3,HIGH);
  period = pulseIn(sensorOut, LOW);
  Serial.print("B= ");
  Serial.print(period);
  Serial.println("  ");
  delay(100);
}
