#include <EEPROM.h>

#define address 1

  Serial.begin(9600);
  // „ﬁœ«— œÂÌ «Ê·ÌÂ »Â Õ«›ŸÂ »«Ìœ ’Ê—  êÌ—œ
  byte firstValue = 0;

void loop(){
  if(Serial.available()){
   char reading = Serial.read();
   byte value = Serial.parseInt(); 
   if ( reading == 'w')
     EEPROM.update(address, value);
   else if ( reading == 'r')
     Serial.println(EEPROM.read(address)); 
   }
}
