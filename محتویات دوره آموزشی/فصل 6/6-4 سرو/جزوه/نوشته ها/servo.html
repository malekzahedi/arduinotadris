#include <Servo.h>
// ساختن شی سرو و اسم گذاری آن
Servo myservo; 

int inputPin = 0;  
int val;    

void setup() {
  // تعریف پین متصل شده به سیم سیگنال موتور
  myservo.attach(9);  
}

void loop() {
  val = analogRead(inputPin);            
  val = map(value, 0, 1023, 0, 180); 
  // نوشتن اندازه زاویه یا سرعت دورانی   
  myservo.write(val);
// تاخیری به مدار می دهیم تا سرو فرصت کند به زاویه مطلوب برسد.  
  delay(15);                           
}
